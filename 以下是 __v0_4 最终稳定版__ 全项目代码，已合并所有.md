```javascript
const BOOLEAN_ATTRS = new Set(['checked', 'disabled', 'readonly', 'selected', 'multiple', 'autofocus', 'required']);

export function getFromStorage(key, defaultValue = null) {
  try { const item = localStorage.getItem(key); return item ? JSON.parse(item) : defaultValue; }
  catch { return defaultValue; }
}

export function setToStorage(key, value) {
  try { localStorage.setItem(key, JSON.stringify(value)); } catch { /* ignore */ }
}

export function isMobile() {
  return window.matchMedia('(max-width: 768px)').matches;
}

export function createElement(tag, attrs = {}, children = []) {
  const el = document.createElement(tag);
  for (const [key, value] of Object.entries(attrs)) {
    if (key === 'className') el.className = value;
    else if (key === 'innerHTML') el.innerHTML = value;
    else if (key === 'textContent') el.textContent = value;
    else if (key.startsWith('on') && typeof value === 'function') el.addEventListener(key.slice(2).toLowerCase(), value);
    else if (key === 'style' && typeof value === 'object') Object.assign(el.style, value);
    else if (key === 'data') for (const [dk, dv] of Object.entries(value)) el.dataset[dk] = dv;
    else if (BOOLEAN_ATTRS.has(key) && typeof value === 'boolean') el[key] = value;
    else el.setAttribute(key, value);
  }
  if (typeof children === 'string') el.innerHTML = children;
  else if (Array.isArray(children)) {
    for (const child of children) {
      if (typeof child === 'string') el.appendChild(document.createTextNode(child));
      else if (child instanceof Node) el.appendChild(child);
    }
  }
  return el;
}
```

```javascript
let currentRequestId = 0;
let activeDropdown = null, activeOverlay = null;

export function initSuggestions(searchBox, inputEl, toggleBtn, onSearch) {
  // 清理旧元素

  overlay.className = 'suggestions-overlay';

  overlay.addEventListener('click', () => closeHandler());
  document.body.appendChild(overlay);

  let debounceTimer, lastQuery = '';

  document.addEventListener('close-suggestions', () => closeHandler());

  inputEl.addEventListener('input', () => {
    const q = inputEl.value.trim();
    if (q === lastQuery) return;
    lastQuery = q;
    clearTimeout(debounceTimer);
    if (!q) { closeHandler(); return; }
    debounceTimer = setTimeout(() => fetchAndShow(q), 300);
  });

  inputEl.addEventListener('keydown', (e) => {
    if (e.key === 'Escape') { closeHandler(); return; }
    const visible = activeDropdown && activeDropdown.style.display === 'block';
    if (e.key === 'ArrowDown' || e.key === 'ArrowUp') {
      if (!visible) return;
      e.preventDefault();
      const items = activeDropdown.querySelectorAll('.suggestion-item');
      selectedIndex = e.key === 'ArrowDown' ? Math.min(selectedIndex + 1, items.length - 1) : Math.max(selectedIndex - 1, -1);
      items.forEach((el, i) => {
        el.style.backgroundColor = i === selectedIndex ? 'var(--color-border)' : '';
        el.classList.toggle('active', i === selectedIndex);
      });
    } else if (e.key === 'Enter') {
      e.preventDefault();
      let text = inputEl.value.trim();
      if (visible && selectedIndex >= 0) {
        const items = activeDropdown.querySelectorAll('.suggestion-item');
        if (items[selectedIndex]) text = items[selectedIndex].textContent;
      }
      closeHandler();
      if (text) { inputEl.value = text; onSearch(text); }
    }
  });

  inputEl.addEventListener('blur', () => {
    setTimeout(() => { if (!document.activeElement || !searchBox.contains(document.activeElement)) closeHandler(); }, 150);
  });

  function fetchAndShow(query) {
    const src = getCurrentSource();
    if (!src || src.parser === 'none') { closeHandler(); return; }
    const reqId = ++currentRequestId;
    const cbName = '__sug_' + Date.now() + '_' + reqId;
    let removed = false;
    window[cbName] = (data) => {
      if (reqId !== currentRequestId) { delete window[cbName]; return; }
      if (!removed) cleanup();
      const sugg = parse(data, src.parser);
      if (reqId === currentRequestId) show(sugg);
      delete window[cbName];
    };
    const url = src.url.replace('{query}', encodeURIComponent(query)).replace('{callback}', cbName);
    const script = document.createElement('script');
    script.src = url;
    script.onerror = () => { delete window[cbName]; cleanup(); closeHandler(); };
    document.body.appendChild(script);
    const timer = setTimeout(() => { if (!removed) { delete window[cbName]; cleanup(); closeHandler(); } }, 5000);
    function cleanup() { removed = true; clearTimeout(timer); script.remove(); }
  }

  function parse(data, parser) {
    try {
      switch (parser) {
        case 'generic_array': if (Array.isArray(data)) { if (data[1] && Array.isArray(data[1])) return data[1].slice(0,8); if (data.every(i => typeof i === 'string')) return data.slice(0,8); } return [];
        case 'baidu': if (data?.s) return data.s.slice(0,8); if (data?.g) return data.g.map(i => i.q).slice(0,8); return [];
        case 'ddg': if (Array.isArray(data)) return data.map(i => i.phrase || i.text || '').filter(Boolean).slice(0,8); return [];
        default: return parse(data, 'generic_array');
      }
    } catch { return []; }
  }

  function show(suggestions) {
    if (!suggestions.length) { closeHandler(); return; }
    currentSuggestions = suggestions; selectedIndex = -1;
    dropdown.innerHTML = '';
    suggestions.forEach(text => {
      const item = document.createElement('div');
      item.className = 'suggestion-item';
      item.textContent = text;
      Object.assign(item.style, { padding:'8px 16px', cursor:'pointer', fontSize:'14px' });

      item.onmousedown = (e) => { e.preventDefault(); closeHandler(); inputEl.value = text; onSearch(text); };
      dropdown.appendChild(item);
    });
    dropdown.style.display = 'block'; activeDropdown = dropdown;
    if (window.innerWidth <= 768) { overlay.style.display = 'block'; activeOverlay = overlay; }
  }

  closeHandler = () => {
    if (activeDropdown) { activeDropdown.style.display = 'none'; activeDropdown.innerHTML = ''; activeDropdown = null; }
    if (activeOverlay) { activeOverlay.style.display = 'none'; activeOverlay = null; }
    currentSuggestions = []; selectedIndex = -1;
  };
}
```

```javascript
  const name = createElement('span', { className:'category-name' }, `${cat.name} (${bms.length})`);
  const actions = createElement('div', { className:'category-actions' });
  if (cat.id !== 'default') {
    actions.append(
      createElement('button', { className:'btn-icon', onclick: (e) => { e.stopPropagation(); const n = prompt('新名称', cat.name); if(n?.trim() && n.trim()!==cat.name) { updateCategory(cat.id, n.trim()); renderMain(); } } }, '✎'),
      createElement('button', { className:'btn-icon', onclick: (e) => { e.stopPropagation(); if(confirm('删除分区？')) { deleteCategory(cat.id); renderMain(); } } }, '✕')
    );
  }
  header.append(expand, name, actions);
  const listDiv = createElement('div', { className:'bookmark-list', style:{display:'none'} });
  header.addEventListener('click', () => {
    const visible = listDiv.style.display === 'block';
    document.querySelectorAll('.bookmark-list').forEach(el => el.style.display = 'none');
    document.querySelectorAll('.expand-icon').forEach(el => el.textContent = '▶');
    if (!visible) { listDiv.style.display = 'block'; expand.textContent = '▼'; if (!listDiv.children.length) renderBmList(listDiv, cat.id); }
  });
  item.append(header, listDiv);
  return item;
}

function renderBmList(container, catId) {
  container.innerHTML = '';
  const bms = getBookmarksByCategory(catId);
  if (!bms.length) container.appendChild(createElement('div', { className:'empty-hint' }, '暂无收藏'));
  else bms.forEach(bm => {
    const row = createElement('div', { className:'bookmark-item' });
    const icon = createElement('img', { className:'bookmark-icon', src: bm.iconUrl || getFavicon(bm.url), onerror: (e) => { e.target.style.display = 'none'; } });
    const info = createElement('div', { className:'bookmark-info' });
    info.append(createElement('a', { href: bm.url, target:'_blank', className:'bookmark-title' }, bm.title), createElement('span', { className:'bookmark-url' }, bm.url));
    const actions = createElement('div', { className:'bookmark-actions' });
    actions.append(
      createElement('button', { className:'btn-icon', onclick: () => showBmForm(null, bm) }, '✎'),
      createElement('button', { className:'btn-icon', onclick: () => { if(confirm('删除？')) { deleteBookmark(bm.id); renderMain(); } } }, '✕')
    );
    row.append(icon, info, actions);
    container.appendChild(row);
  });
  container.appendChild(createElement('button', { className:'btn-add-bookmark', onclick: () => showBmForm(catId) }, '+ 添加网站'));
}

function handleSearch(query, resultsDiv) {
  const catList = document.querySelector('.category-list');
  if (!query.trim()) {
    if (resultsDiv) resultsDiv.style.display = 'none';
    if (catList) catList.style.display = '';
    return;
  }
  if (catList) catList.style.display = 'none';
  if (resultsDiv) resultsDiv.style.display = 'block';
  const results = searchBookmarks(query.trim());
  resultsDiv.innerHTML = '';
  if (!results.length) resultsDiv.appendChild(createElement('div', { className:'empty-hint' }, '无结果'));
  else results.forEach(bm => {
    const cat = getCategoryById(bm.categoryId);
    const item = createElement('div', { className:'search-result-item' });
    const icon = createElement('img', { className:'bookmark-icon small', src: bm.iconUrl || getFavicon(bm.url), onerror: (e) => { e.target.style.display='none'; } });
    const info = createElement('div', { className:'bookmark-info' });
    info.append(createElement('a', { href: bm.url, target:'_blank', className:'bookmark-title' }, bm.title), createElement('span', { className:'bookmark-category' }, `分类: ${cat?.name || '未分类'}`));
    item.append(icon, info); resultsDiv.appendChild(item);
  });
}

function showBmForm(catId, editData) {
  const content = document.getElementById('bmContent'); content.innerHTML = '';
  const form = createElement('div', { className:'bookmark-form' });
  const titleInp = createElement('input', { type:'text', id:'bmTitle', placeholder:'网站名称', value: editData?.title || '' });
  const urlInp = createElement('input', { type:'text', id:'bmUrl', placeholder:'网站URL', value: editData?.url || '' });
  const iconInp = createElement('input', { type:'text', id:'bmIcon', placeholder:'图标URL (可选)', value: editData?.iconUrl || '' });
  const catSel = createElement('select', { id:'bmCategory' });
  getAllCategories().forEach(c => catSel.appendChild(createElement('option', { value: c.id, selected: c.id === (editData?.categoryId || catId || 'default') }, c.name)));
  const save = () => {
    const title = titleInp.value.trim(), url = (urlInp.value.trim().startsWith('http') ? '' : 'https://') + urlInp.value.trim(), icon = iconInp.value.trim(), categoryId = catSel.value;
    if (!title || !url) return alert('必填');
    if (editData) updateBookmark(editData.id, { title, url, iconUrl: icon, categoryId });
    else addBookmark({ title, url, iconUrl: icon, categoryId });
    renderMain();
  };
  form.append(createElement('label', {}, '名称'), titleInp, createElement('label', {}, 'URL'), urlInp, createElement('label', {}, '图标URL'), iconInp, createElement('label', {}, '分类'), catSel);
  form.appendChild(createElement('div', { className:'form-actions' }, [
    createElement('button', { className:'btn-save', onclick: save }, '保存'),
    createElement('button', { className:'btn-cancel', onclick: renderMain }, '取消')
  ]));
  content.appendChild(form);
}
```
